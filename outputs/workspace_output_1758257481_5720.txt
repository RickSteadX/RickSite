     1	using System;
     2	using System.Text.Json;
     3	
     4	namespace PersonalWebsite.Client.Models
     5	{
     6	    public class PageComponent
     7	    {
     8	        public int Id { get; set; }
     9	        public int PageId { get; set; }
    10	        public string ComponentType { get; set; }
    11	        public string Content { get; set; }

    13	        public int Order { get; set; }
    14	        public string PropertiesJson { get; set; }
    15	        
    16	        // Helper method to get properties as dynamic object
    17	        public T GetProperties<T>() where T : class, new()
    18	        {
    19	            if (string.IsNullOrEmpty(PropertiesJson))
    20	                return new T();
    21	                
    22	            return JsonSerializer.Deserialize<T>(PropertiesJson);
    23	        }
    24	        
    25	        // Helper method to set properties from object
    26	        public void SetProperties<T>(T properties) where T : class
    27	        {
    28	            PropertiesJson = JsonSerializer.Serialize(properties);
    29	        }
    30	    }
    31	}
